예제 코드로 재질 할당 익히기: 단계별 프로세스

재질을 생성하고 오브젝트에 할당하는 전체 과정을 코드로 따라가 봅시다.

[실행 코드]

import maya.cmds as mc

# 준비: 재질을 할당할 오브젝트를 미리 생성하고 선택해둡니다.
my_obj = mc.polySphere(name="mySphere")[0]
mc.select(my_obj)

# --- 1단계: 재질(Material) 생성 ---
red_mat = mc.shadingNode('blinn', asShader=True, name='red_mat')
mc.setAttr(red_mat + '.color', 1, 0, 0, type='double3')

# --- 2단계: 쉐이딩 그룹(SG) 생성 ---
# renderable=True, noSurfaceShader=True 는 쉐이딩 그룹을 만드는 공식입니다.
red_mat_sg = mc.sets(renderable=True, noSurfaceShader=True, empty=True, name='red_matSG')

# --- 3단계: 재질과 쉐이딩 그룹 연결 ---
# red_mat의 outColor를 red_mat_sg의 surfaceShader 속성에 연결합니다.
mc.connectAttr(red_mat + '.outColor', red_mat_sg + '.surfaceShader', force=True)

# --- 4단계: 선택한 오브젝트에 쉐이딩 그룹 할당 ---
# red_mat_sg 세트(set)에 선택한 오브젝트(my_obj)를 강제로 포함(forceElement)시킵니다.
mc.sets(my_obj, edit=True, forceElement=red_mat_sg)

print(f"'{my_obj}'에 '{red_mat}' 재질을 할당했습니다.")

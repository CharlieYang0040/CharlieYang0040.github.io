OS

os 심화: 환경변수, 파일 탐색, 경로 정규화

현업에서 자주 겪는 문제를 다룹니다. 환경변수 안전 사용, 대규모 폴더 순회(os.walk), 플랫폼 간 경로 안정화(abspath, normpath, relpath)를 연습합니다.

[기본 예제]

import os

# 환경변수 읽기 (기본값 포함)
mode = os.environ.get("MY_APP_MODE", "dev")
print("MODE:", mode)

# 경로 다루기: join → normpath → 분해
home = os.path.expanduser("~")
project = os.path.normpath(os.path.join(home, "my_project", "assets"))
print("PROJECT:", project)
print("DIR:", os.path.dirname(project))
print("BASE:", os.path.basename(project))

# 상대경로로 보기 좋게 표시
config = os.path.join(project, "config.json")
print("REL:", os.path.relpath(config, start=os.path.dirname(project)))

[실습]

지정한 루트(ROOT) 아래의 모든 .ma/.mb 파일을 찾아, 현재 폴더에 report.txt로 상대경로와 바이트 크기를 크기 내림차순으로 기록하세요.

해답:

import os
from operator import itemgetter

ROOT = os.path.expanduser("~/my_project")
ROOT = os.path.normpath(ROOT)
items = []

for cur_dir, dirnames, filenames in os.walk(ROOT):
	# 숨김 폴더 제외
	dirnames[:] = [d for d in dirnames if not d.startswith('.')]
	for name in filenames:
		lower = name.lower()
		if lower.endswith('.ma') or lower.endswith('.mb'):
			abs_path = os.path.join(cur_dir, name)
			size = os.path.getsize(abs_path)
			rel = os.path.relpath(abs_path, start=ROOT)
			items.append((rel, size))

items.sort(key=itemgetter(1), reverse=True)

with open('report.txt', 'w', encoding='utf-8') as f:
	for rel, size in items:
		f.write(f"{rel}\t{size}\n")

print("생성됨: report.txt, 항목 수:", len(items))

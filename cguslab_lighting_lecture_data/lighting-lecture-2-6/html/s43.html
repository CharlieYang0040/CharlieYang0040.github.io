최종 종합 실습: 라이트 생성 후 오브젝트 바라보게 만들기

지금까지 배운 모든 것을 활용합니다: 매개변수, return, 그리고 다른 함수 호출

[실행 코드]

import maya.cmds as mc

# 2교시 예제를 수정: 생성한 라이트의 트랜스폼 노드 이름을 return 하도록 변경
def create_spotlight_at_center(target_object):
    bbox = mc.xform(target_object, q=True, bb=True, ws=True)
    center_x = (bbox[0] + bbox[3]) / 2.0
    center_y = (bbox[1] + bbox[4]) / 2.0
    center_z = (bbox[2] + bbox[5]) / 2.0

    # 스포트라이트를 만들고, 그 결과(트랜스폼, 쉐입 노드 이름 리스트)를 받습니다.
    new_light_nodes = mc.spotLight(position=[center_x, center_y + 10, center_z])

    # 트랜스폼 노드 이름(리스트의 첫번째 항목)을 돌려줍니다.
    return new_light_nodes[0]

def aim_light_at_object(light, target):
    # light가 target을 항상 바라보도록 aimConstraint를 설정합니다.
    mc.aimConstraint(target, light)
    print(f"{light}가 {target}을(를) 바라보도록 설정했습니다.")

# --- 최종 실행 코드 ---
# 1. 목표 오브젝트 설정 (미리 만들어두거나, 코드로 생성)
target = "pSphere1"
if not mc.objExists(target):
    mc.polySphere(name=target)

# 2. 목표 오브젝트 위치에 스포트라이트를 생성하고, 그 이름을 new_light_name 변수에 저장
new_light_name = create_spotlight_at_center(target)

# 3. 방금 생성한 라이트가 목표 오브젝트를 바라보도록 설정
aim_light_at_object(new_light_name, target)

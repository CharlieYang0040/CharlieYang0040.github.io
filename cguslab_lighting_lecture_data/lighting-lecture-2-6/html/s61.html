함수 2: 3점 조명 자동 생성

선택한 오브젝트의 크기를 기준으로 Key, Fill, Rim 라이트의 위치와 강도를 자동으로 계산하여 배치합니다.

[함수 2: 3점 조명 생성]

import maya.cmds as mc

def create_three_point_lights(target_object):
    """
    선택한 오브젝트를 기준으로 Key, Fill, Rim 라이트를 생성하는 함수
    """
    # 1. target_object의 바운딩 박스(Bounding Box) 정보를 가져옵니다.
    bbox = mc.xform(target_object, q=True, bb=True, worldSpace=True)
    
    # 2. 오브젝트의 대략적인 크기를 계산하여 라이트 배치에 활용합니다.
    size_x = bbox[3] - bbox[0]
    size_y = bbox[4] - bbox[1]
    size_z = bbox[5] - bbox[2]
    object_size = max(size_x, size_y, size_z)

    # 3. Key Light (Area Light) 생성 및 배치
    key_light = mc.shadingNode('areaLight', asLight=True, name="Key_Light")
    key_light_transform = mc.listRelatives(key_light, parent=True)[0]
    mc.move(-object_size, object_size, object_size, key_light_transform)
    mc.setAttr(key_light + ".intensity", object_size * 40) # 크기에 비례한 강도

    # 4. Fill Light (Area Light) 생성 및 배치
    fill_light = mc.shadingNode('areaLight', asLight=True, name="Fill_Light")
    fill_light_transform = mc.listRelatives(fill_light, parent=True)[0]
    mc.move(object_size, object_size * 0.5, object_size, fill_light_transform)
    mc.setAttr(fill_light + ".intensity", object_size * 10)

    # 5. Rim Light (Spot Light) 생성 및 배치
    rim_light_transform = mc.spotLight(name="Rim_Light")
    mc.move(0, object_size, -object_size * 1.5, rim_light_transform)
    mc.setAttr(rim_light_transform + ".intensity", object_size * 30)

    # 6. 모든 라이트가 target_object를 바라보도록 Aim Constraint 설정
    mc.aimConstraint(target_object, key_light_transform)
    mc.aimConstraint(target_object, fill_light_transform)
    mc.aimConstraint(target_object, rim_light_transform)

    print("3점 조명을 생성했습니다.")

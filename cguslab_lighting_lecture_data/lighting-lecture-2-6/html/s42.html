예제 코드로 Return 익히기 2: 트랜스폼 노드로 쉐입 노드 찾아 돌려주기

매우 유용한 유틸리티 함수입니다. 어떤 트랜스폼 노드든, 그 아래의 쉐입 노드를 정확히 찾아줍니다.

[실행 코드]

import maya.cmds as mc

def get_shape_node(transform_node):
    # listRelatives 명령어로 쉐입 노드를 찾습니다.
    shapes = mc.listRelatives(transform_node, shapes=True)
    
    if shapes: # 쉐입 노드 리스트가 비어있지 않다면 (쉐입 노드가 존재한다면)
        return shapes[0] # 첫번째 쉐입 노드의 이름을 돌려준다.
    else: # 쉐입 노드가 없다면
        return None # 없다는 의미로 'None'을 돌려준다.

# --- 사용법 ---
# 1. 씬에 pointLight를 하나 만들면 'pointLight1' 트랜스폼이 생성됩니다.
# 2. 아래 코드를 실행해보세요.

light_transform = "pointLight1"
light_shape = get_shape_node(light_transform)

if light_shape:
    print(f"{light_transform}의 쉐입 노드는: {light_shape}") # 출력: pointLight1의 쉐입 노드는: pointLightShape1
    mc.setAttr(light_shape + ".intensity", 5) # 찾은 쉐입노드의 속성 제어
else:
    print(f"{light_transform}에서 쉐입 노드를 찾을 수 없습니다.")

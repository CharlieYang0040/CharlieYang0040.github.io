예제 3: 3주차 스크립트를 UI로 만들기

지난 주에 완성한 복잡한 스크립트를 클릭 한 번으로 실행되는 간단한 툴로 포장합니다.

[실행 코드]

import maya.cmds as mc

# --- 3주차에 만들었던 함수들을 여기에 붙여넣습니다 ---
# apply_material_by_type 함수가 정의되어 있다고 가정합니다.
# def apply_material_by_type(objects, mat_name, mat_type, color_rgb):
#     # ... (3주차 심화 문제에서 만든 코드 내용) ...
#     pass # 실제 코드로 채워주세요

def run_auto_material_assigner(*args):
    """
    버튼 클릭 시 실행될 메인 함수. 3주차 스크립트의 실행 로직을 담고 있습니다.
    """
    USE_VRAY = False # 여기서는 간단하게 Blinn 재질만 할당하도록 설정
    print(f"===== 재질 자동 할당을 시작합니다 (V-Ray 모드: {USE_VRAY}) =====")
    
    window_objs = mc.ls('*_window_*', type='transform')
    if window_objs:
        apply_material_by_type(window_objs, 'Glass_Mat', 'blinn', [0.2, 0.8, 1.0])
        if mc.objExists('Glass_Mat'):
            mc.setAttr('Glass_Mat.transparency', 0.8, 0.8, 0.8, type='double3')

    frame_objs = mc.ls('*_frame_*', type='transform')
    if frame_objs:
        apply_material_by_type(frame_objs, 'Metal_Frame_Mat', 'blinn', [0.2, 0.2, 0.2])

    wall_objs = mc.ls('*_wall_*', type='transform')
    if wall_objs:
        apply_material_by_type(wall_objs, 'Concrete_Wall_Mat', 'lambert', [0.5, 0.5, 0.5])
    
    print("===== 모든 작업이 완료되었습니다 =====")
# -------------------------------------------------

if mc.window("materialAssignerUI", exists=True):
    mc.deleteUI("materialAssignerUI")

mc.window("materialAssignerUI", title="재질 자동 할당기")
# adjustableColumn=True : 창 크기를 조절하면 버튼 너비도 함께 조절됩니다.
mc.columnLayout(adjustableColumn=True) 

mc.button(label="씬 전체 재질 자동 할당 실행", command=run_auto_material_assigner, height=50)

mc.showWindow("materialAssignerUI")

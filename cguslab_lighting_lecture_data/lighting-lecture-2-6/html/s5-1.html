shutil 심화: 폴더/파일 관리와 예외 처리

파일 시스템 작업을 자동화합니다. 대용량 폴더 복사(copytree), 특정 조건 파일 이동, 작업 중 발생하는 권한 및 경로 오류 처리(ignore_errors)를 다룹니다.

[기본 예제]

import os, shutil
from datetime import datetime

src = "textures"
clean = os.path.join("backup", "textures_clean")
os.makedirs(os.path.dirname(clean), exist_ok=True)

# 1) 제외 패턴 적용 복사
shutil.copytree(
	src, clean,
	dirs_exist_ok=True,
	ignore=shutil.ignore_patterns("*.tmp", "*.bak", "Thumbs.db", "__pycache__")
)

# 2) 아카이브 생성 (zip)
stamp = datetime.now().strftime("%Y%m%d")
archive_base = os.path.join("backup", f"textures_{stamp}")
archive_path = shutil.make_archive(archive_base, "zip", root_dir="backup", base_dir="textures_clean")
print("아카이브:", archive_path)

# 3) 용량 확인
usage = shutil.disk_usage(os.getcwd())
print("남은 용량(GB):", round(usage.free/1024**3, 2))

[실습]

현재 폴더의 assets를 제외 패턴(*.tmp, __pycache__)로 복사 후, backup/assets_YYYYMMDD.zip으로 압축하고, 임시 폴더에 해제하여 파일 수가 원본과 동일한지 검증하세요.

해답:

import os, shutil
from datetime import datetime

src = "assets"
clean = os.path.join("backup", "assets_clean")
os.makedirs(os.path.dirname(clean), exist_ok=True)
shutil.copytree(src, clean, dirs_exist_ok=True, ignore=shutil.ignore_patterns("*.tmp", "__pycache__"))

stamp = datetime.now().strftime("%Y%m%d")
archive = shutil.make_archive(os.path.join("backup", f"assets_{stamp}"), "zip", root_dir="backup", base_dir="assets_clean")

# 검증: 임시 해제 후 파일 수 비교
tmp = os.path.join("backup", "_verify")
if os.path.exists(tmp):
	shutil.rmtree(tmp)
shutil.unpack_archive(archive, tmp)

# 파일 수 세기
def count_files(d):
	n = 0
	for _, _, files in os.walk(d):
		n += len(files)
	return n

print("원본:", count_files(clean), "해제본:", count_files(tmp))

mc.ls() 마스터하기: 예제 3

이름 규칙으로 검색하기 (와일드카드 * 활용)

와일드카드(*)는 '어떤 글자든 상관없다'는 의미의 특수문자입니다. 이름의 일부만으로 원하는 오브젝트 그룹을 정확하게 찾아낼 수 있어 매우 강력합니다.

[특정 단어로 시작하는 오브젝트 검색]

import maya.cmds as mc

# 이름이 'pCube'로 시작하는 모든 오브젝트를 검색
# 예: pCube1, pCube_body, pCube_arm 등
cube_list = mc.ls('pCube*')

print("===== 'pCube'로 시작하는 오브젝트 목록 =====")
for cube in cube_list:
    print(cube)
print("=========================================")

[특정 단어를 포함하는 오브젝트 검색]

import maya.cmds as mc

# 이름에 '_Lgt'가 포함된 모든 오브젝트를 검색
# 예: Key_Lgt, Rim_Lgt_01 등
light_list_by_name = mc.ls('*_Lgt*')

print("\n===== 이름에 '_Lgt'가 포함된 오브젝트 목록 =====")
for light in light_list_by_name:
    print(light)
print("==============================================")

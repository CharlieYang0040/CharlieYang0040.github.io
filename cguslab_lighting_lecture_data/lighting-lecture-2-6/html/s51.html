예제: 생성부터 할당까지 한번에! 범용 함수 만들기

이전 슬라이드의 복잡한 과정을 하나의 함수로 묶어 재사용성을 극대화합니다.

이 함수는 재질의 종류를 나타내는 mat_type 매개변수 덕분에, Blinn, Lambert 같은 기본 재질은 물론 'VRayMtl' 같은 외부 렌더러 재질까지 모두 처리할 수 있는 매우 유연하고 실용적인 함수입니다.

[실행 코드]

import maya.cmds as mc

def assign_new_material_to_selection(mat_name, mat_type, color_rgb):
    selection = mc.ls(sl=True)
    if not selection:
        print("재질을 할당할 오브젝트를 먼저 선택해주세요.")
        return

    # 1. 재질 생성 (매개변수 mat_type 사용)
    material = mc.shadingNode(mat_type, asShader=True, name=mat_name)
    mc.setAttr(material + '.color', color_rgb[0], color_rgb[1], color_rgb[2], type='double3')

    # 2. 쉐이딩 그룹 생성 및 연결
    sg = mc.sets(renderable=True, noSurfaceShader=True, empty=True, name=mat_name + 'SG')
    mc.connectAttr(material + '.outColor', sg + '.surfaceShader', force=True)

    # 3. 선택한 오브젝트에 할당
    mc.sets(selection, edit=True, forceElement=sg)
    print(f"선택한 오브젝트에 '{mat_name}' 재질({mat_type})을 할당했습니다.")

# --- 사용법 ---
# 사용법 1: 선택한 오브젝트에 'gold_mat' (blinn) 재질 할당하기
# assign_new_material_to_selection('gold_mat', 'blinn', [1.0, 0.8, 0.2])

# 사용법 2: 선택한 오브젝트에 'vray_plastic_mat' (VRayMtl) 재질 할당하기
# assign_new_material_to_selection('vray_plastic_mat', 'VRayMtl', [0.8, 0.1, 0.1])

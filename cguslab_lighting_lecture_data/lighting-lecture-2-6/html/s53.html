3교시: 실전! 이름으로 도시 건물 재질 자동 할당하기

수십 개의 건물 모델링 이름 규칙을 분석하여 V-Ray 또는 Maya 기본 재질을 자동으로 할당합니다.

[최종 스크립트: 재질 자동 할당기]

import maya.cmds as mc

# (2교시 심화 문제에서 만든 apply_material_by_type 함수가 이 코드 위에 있다고 가정합니다)
# def apply_material_by_type(objects, mat_name, mat_type, color_rgb): ...

# ===== 설정 =====
USE_VRAY = True # 이 값을 True로 바꾸면 V-Ray 재질로, False로 바꾸면 Blinn 재질로 할당됩니다.
# ================

print(f"===== 재질 자동 할당을 시작합니다 (V-Ray 모드: {USE_VRAY}) =====")

# 1. 창문 오브젝트: V-Ray 사용 시 유리 재질, 아니면 투명한 Blinn 할당
window_objs = mc.ls('*_window_*', type='transform')
if window_objs:
    if USE_VRAY:
        apply_material_by_type(window_objs, 'Glass_VRay_Mat', 'VRayMtl', [0, 0, 0])
        if mc.objExists('Glass_VRay_Mat'):
            mc.setAttr('Glass_VRay_Mat.reflectionColor', 1, 1, 1, type='double3')
            mc.setAttr('Glass_VRay_Mat.refractionColor', 1, 1, 1, type='double3')
            mc.setAttr('Glass_VRay_Mat.refractionIOR', 1.52) # 유리의 굴절률
    else:
        apply_material_by_type(window_objs, 'Glass_Mat', 'blinn', [0.2, 0.8, 1.0])
        if mc.objExists('Glass_Mat'):
            mc.setAttr('Glass_Mat.transparency', 0.8, 0.8, 0.8, type='double3')

# 2. 창틀 오브젝트: V-Ray 사용 시 금속 재질, 아니면 Blinn 할당
frame_objs = mc.ls('*_frame_*', type='transform')
if frame_objs:
    if USE_VRAY:
        apply_material_by_type(frame_objs, 'Metal_Frame_VRay_Mat', 'VRayMtl', [0.1, 0.1, 0.1])
        if mc.objExists('Metal_Frame_VRay_Mat'):
            mc.setAttr('Metal_Frame_VRay_Mat.reflectionColor', 0.9, 0.9, 0.9, type='double3')
            mc.setAttr('Metal_Frame_VRay_Mat.reflectionGlossiness', 0.8)
    else:
        apply_material_by_type(frame_objs, 'Metal_Frame_Mat', 'blinn', [0.2, 0.2, 0.2])

# 3. 벽 오브젝트 (V-Ray/Lambert는 비슷하므로 Lambert로 통일)
wall_objs = mc.ls('*_wall_*', type='transform')
if wall_objs:
    apply_material_by_type(wall_objs, 'Concrete_Wall_Mat', 'lambert', [0.5, 0.5, 0.5])

print("===== 모든 작업이 완료되었습니다 =====")

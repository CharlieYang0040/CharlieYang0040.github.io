2단계: 최종 UI 제작 및 연결

앞서 만든 마스터 함수를 버튼에 연결하고, 간단한 안내 문구를 추가하여 툴을 완성합니다.

[최종 툴 전체 코드]

import maya.cmds as mc

# (2교시 함수들과 run_turntable_generator 함수가 이 코드 위에 모두 포함되어야 합니다)
# def create_backdrop(...): ...
# def create_three_point_lights(...): ...
# def create_turntable_camera(...): ...
# def run_turntable_generator(...): ...


# --- 최종 UI 제작 ---
UI_ID = "lookdevTurntableUI"
if mc.window(UI_ID, exists=True):
    mc.deleteUI(UI_ID)

mc.window(UI_ID, title="룩뎁 턴테이블 생성기 v1.0", widthHeight=(350, 150))
mc.columnLayout(adjustableColumn=True, rowSpacing=10, margin=10)

# 사용자 안내를 위한 텍스트 추가
mc.text(label="1. 턴테이블을 만들 오브젝트를 선택하세요.")
mc.text(label="2. 아래 버튼을 클릭하세요.")

# 시각적 구분을 위한 구분선 추가
mc.separator(height=20, style='in')

# 핵심 기능을 실행할 메인 버튼
mc.button(label="턴테이블 씬 생성 실행", 
          command=run_turntable_generator, 
          height=50, 
          backgroundColor=[0.2, 0.5, 0.3]) # 버튼 색상 변경

mc.showWindow(UI_ID)

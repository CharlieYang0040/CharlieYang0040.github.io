실전 구현 코드

Python API
# UsdShade Material 생성
material = UsdShade.Material.Define(stage, '/Root/Materials/MyMaterial')
# Surface Shader 생성
shader = UsdShade.Shader.Define(stage, '/Root/Materials/MyMaterial/PbrShader')
shader.CreateIdAttr('UsdPreviewSurface')
# 쉐이더 파라미터 설정
shader.CreateInput('diffuseColor', Sdf.ValueTypeNames.Color3f).Set(color)
shader.CreateInput('roughness', Sdf.ValueTypeNames.Float).Set(0.4)
shader.CreateInput('metallic', Sdf.ValueTypeNames.Float).Set(0.1)
# Material 출력에 연결
material.CreateSurfaceOutput().ConnectToSource(shader, 'surface')
# MaterialX 변환 설정
mtlx = UsdShade.MaterialBindingAPI(prim)
# 바인딩 API 활용
mtlx.Bind(material)
# 프리뷰 설정
material.CreateInput('useSpecularWorkflow', Sdf.ValueTypeNames.Int).Set(1)
